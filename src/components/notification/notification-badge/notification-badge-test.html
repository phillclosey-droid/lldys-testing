Implementation Guide

Complete HTML structure with all dependencies.

<!DOCTYPE html>
<html lang="en">
<head>
    <!-- 1. Design Tokens (REQUIRED) -->
    <link rel="stylesheet" href="path/to/cancara-tokens.css">

    <!-- 2. NotificationBadge CSS -->
    <link rel="stylesheet" href="path/to/NotificationBadge.vanilla.css">
</head>
<body data-theme="light">

    <!-- Default badge (red) -->
    <div class="notification-badge">
        <div class="notification-badge-text">12</div>
    </div>

    <!-- Offers badge (green) -->
    <div class="notification-badge notification-badge-offers">
        <div class="notification-badge-text">3</div>
    </div>

    <!-- Positioned on icon -->
    <div style="position: relative;">
        <!-- Your icon -->
        <div class="notification-badge" style="position: absolute; top: -4px; right: -4px;">
            <div class="notification-badge-text">5</div>
        </div>
    </div>

</body>
</html>
